import tkinter as tk
from tkinter import ttk
from tkinter import scrolledtext

win = tk.Tk()
win.title("Radio Button and ScrolledText Example")
win.geometry("400x350")

tk.Label(win, text="Enter your Name:").pack(pady=5)
entry_name = tk.Entry(win, width=30)
entry_name.pack(pady=5)

tk.Label(win, text="Enter your Address:").pack(pady=5)
text_address = tk.Text(win, height=5, width=30)
text_address.pack(pady=5)

tk.Label(win, text="Choose the Country:").pack(pady=5)
country = ttk.Combobox(win, values=["USA", "India", "UK", "China", "Singapore"], state="readonly")
country.pack(pady=5)

tk.Label(win, text="Select your Gender:").pack(pady=5)
gender_var = tk.StringVar(value="Male")

tk.Radiobutton(win, text="Male", variable=gender_var, value="Male").pack(anchor="w")
tk.Radiobutton(win, text="Female", variable=gender_var, value="Female").pack(anchor="w")
tk.Radiobutton(win, text="Other", variable=gender_var, value="Other").pack(anchor="w")

tk.Label(win, text="Write your Comments:").pack(pady=10)
text_area = scrolledtext.ScrolledText(win, width=40, height=8, wrap=tk.WORD)
text_area.pack()

def submit():
    name = entry_name.get()
    address = text_address.get("1.0", tk.END).strip()
    selected_country = country.get()
    gender = gender_var.get()
    comments = text_area.get("1.0", tk.END).strip()
    print(f"Name: {name}\nAddress: {address}\nCountry: {selected_country}\nGender: {gender}\nComments: {comments}")

tk.Button(win, text="Submit", command=submit).pack(pady=15)

win.mainloop()
